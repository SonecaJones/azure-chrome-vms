# AZURE CLI CRIAR MAQUINAS VMSS
az vmss create \
  --resource-group dpcrobos \
  --name VMSSRoboDPC \
  --orchestration-mode Flexible \
  --image MinhaImagemVM \
  --instance-count 50 \
  --vm-sku Standard_D2s_v3 \
  --priority Spot \
  --eviction-policy Delete \
  --max-price -1 \
  --admin-username fabricio \
  --generate-ssh-keys \
  --public-ip-per-vm

# LISTAR TODOS OS INSTANCES
az vmss list-instances \
  --resource-group MEU_RG \
  --name MeuScaleSet \
  --query "[].{id:instanceId}" \
  -o table

# LISTAR TODOS OS IPS
az vmss list-instance-public-ips \
  --resource-group MEU_RG \
  --name MeuScaleSet \
  -o table

# STOP + DEALLOCATE TODOS OS INSTANCES
az vmss deallocate \
  --resource-group MEU_RG \
  --name MeuScaleSet
  

# START EM TODOS OS INSTANCES
az vmss start \
  --resource-group MEU_RG \
  --name MeuScaleSet
